function hmToMinutes(hhmm) {
    const [h, m] = hhmm.split(":").map(Number);
    return h * 60 + m;
}

function getMinutesInZone(timeZone) {
    const parts = new Intl.DateTimeFormat("en-GB", {
        timeZone,
        hour: "2-digit",
        minute: "2-digit",
        hour12: false
    }).formatToParts(new Date());

    const hour = Number(parts.find(p => p.type === "hour")?.value || 0);
    const minute = Number(parts.find(p => p.type === "minute")?.value || 0);
    return hour * 60 + minute;
}

function getMarketPhase(nowMin, openMin, closeMin, preOpenMinutes = 30) {
    if (nowMin >= openMin && nowMin < closeMin) return "open";
    if (nowMin >= openMin - preOpenMinutes && nowMin < openMin) return "pre";
    return "closed";
}

function updateMarketPressure(data) {

    const pressureHud = document.getElementById("pressureHud");
    const pressureValue = document.getElementById("pressureValue");
    const pressureSub = document.getElementById("pressureSub");

    if (!pressureHud || !pressureValue || !pressureSub) return;

    const s1h = data?.state?.["1h"] || {};
    const s5m = data?.state?.["5m"] || {};
    const s1m = data?.state?.["1m"] || {};
    const bias = data?.bias || {};

    let bull = 0;
    let bear = 0;

    /* MAGNET */
    if (s1h.magnet?.direction === "up") bull += 2;
    if (s1h.magnet?.direction === "down") bear += 2;

    if (s5m.magnet?.direction === "up") bull += 1;
    if (s5m.magnet?.direction === "down") bear += 1;

    /* VACUUM */
    if (s1h.vacuum?.vacuum) {
        if (s1h.vacuum.direction === "up") bull += 2;
        if (s1h.vacuum.direction === "down") bear += 2;
    }

    if (s5m.vacuum?.vacuum) {
        if (s5m.vacuum.direction === "up") bull += 1;
        if (s5m.vacuum.direction === "down") bear += 1;
    }

    /* TRAPS */
    if (s1h.trap?.trap) {
        if (s1h.trap.type === "bear_trap") bull += 3;
        if (s1h.trap.type === "bull_trap") bear += 3;
    }

    if (s5m.trap?.trap) {
        if (s5m.trap.type === "bear_trap") bull += 2;
        if (s5m.trap.type === "bull_trap") bear += 2;
    }

    /* SWEEP exhaustion */
    if (s1m.sweep?.sweep_high) bear += 1;
    if (s1m.sweep?.sweep_low) bull += 1;

    /* COMPRESSION = buildup */
    const compression =
        s1h.compression?.compression ||
        s5m.compression?.compression;

    /* BIAS */
    if (bias?.bias === "bullish") bull += 2;
    if (bias?.bias === "bearish") bear += 2;

    let value = "NEUTRAL";
    let sub = "Balanced market";
    let cls = "pressure-neutral";

    if (bull >= bear + 3) {
        value = "BULLISH";
        sub = "Buy pressure dominant";
        cls = "pressure-bull";
    }
    else if (bear >= bull + 3) {
        value = "BEARISH";
        sub = "Sell pressure dominant";
        cls = "pressure-bear";
    }
    else if (compression) {
        value = "BUILDUP";
        sub = "Energy building";
        cls = "pressure-build";
    }
    else if (bull > 0 || bear > 0) {
        value = "EXHAUSTION";
        sub = "Liquidity event detected";
        cls = "pressure-build";
    }

    pressureHud.classList.remove(
        "pressure-bull",
        "pressure-bear",
        "pressure-build",
        "pressure-neutral"
    );

    pressureHud.classList.add(cls);

    pressureValue.textContent = value;
    pressureSub.textContent = sub;
}

async function loadBTC() {
    try {
        const r = await fetch("btc_data.json?v=" + Date.now(), { cache: "no-store" });
        if (!r.ok) throw new Error("HTTP " + r.status);

        const data = await r.json();

        document.getElementById("updatedUtc").textContent = data.meta?.updated_utc || "—";

        const has4h = (data.series?.["4h"] || []).length > 0;
        document.getElementById("marketState").textContent = has4h ? "Activo" : "Sin datos";

        drawChart("btc4h", data.series?.["4h"] || [], data.state?.["4h"]?.liquidity || null);
        drawChart("btc1h", data.series?.["1h"] || [], data.state?.["1h"]?.liquidity || null);
        drawChart("btc5m", data.series?.["5m"] || [], data.state?.["5m"]?.liquidity || null);
        drawChart("btc1m", data.series?.["1m"] || [], data.state?.["1m"]?.liquidity || null);

        renderLiquidity(data.state || {});
        renderSignals(data.state || {});
        renderBias(data.bias || null);
        updateTradeHud(data);
        updateMarketPressure(data);
        updateSmartMoney(data);
        renderDebug(data);

    } catch (err) {
        console.error("Error loading BTC data:", err);
        document.getElementById("marketState").textContent = "Error";
        document.getElementById("signalsBox").innerHTML = "No se pudieron cargar las señales.";
        document.getElementById("liquidityBox").innerHTML = "No se pudieron cargar los niveles de liquidez.";
        document.getElementById("debugBox").innerHTML = "Error de carga: " + err.message;
    }
}

function updateTradeHud(data) {
    const priceEl = document.getElementById("priceText");
    const diffEl = document.getElementById("diffText");
    const targetEl = document.getElementById("targetHudText");
    const targetBox = document.getElementById("targetHud");

    const dotGreen = document.getElementById("dotGreen");
    const dotAmber = document.getElementById("dotAmber");
    const dotRed = document.getElementById("dotRed");

    if (!priceEl || !diffEl || !targetEl || !targetBox) return;

    const series1m = data?.series?.["1m"] || [];
    const lastPoint = series1m.length ? series1m[series1m.length - 1] : null;

    const price = lastPoint?.price ?? null;
    const ema21 = lastPoint?.ema21 ?? null;
    const ema50 = lastPoint?.ema50 ?? null;

    const bias = data?.bias || null;
    const target = bias?.target ?? null;

    priceEl.textContent = price !== null
        ? `$${Number(price).toLocaleString(undefined, { maximumFractionDigits: 2 })}`
        : "$---";

    targetEl.textContent = target !== null
        ? Number(target).toLocaleString(undefined, { maximumFractionDigits: 2 })
        : "---";

    if (price !== null && target !== null) {
        const diff = Math.abs(price - target);
        diffEl.textContent = `${diff.toLocaleString(undefined, {
            minimumFractionDigits: 2,
            maximumFractionDigits: 2
        })}$`;
    } else {
        diffEl.textContent = "0.00$";
    }

    dotGreen.classList.remove("active");
    dotAmber.classList.remove("active");
    dotRed.classList.remove("active");

    let state = "neutral";

    if (price !== null && ema21 !== null && ema50 !== null) {
        if (price > ema21 && ema21 >= ema50) {
            state = "bull";
        } else if (price < ema21 && ema21 <= ema50) {
            state = "bear";
        }
    }

    if (state === "bull") {
        dotGreen.classList.add("active");
    } else if (state === "bear") {
        dotRed.classList.add("active");
    } else {
        dotAmber.classList.add("active");
    }

    targetBox.classList.remove("target-bull", "target-bear", "target-neutral");

    if (bias?.bias === "bullish") {
        targetBox.classList.add("target-bull");
    } else if (bias?.bias === "bearish") {
        targetBox.classList.add("target-bear");
    } else {
        targetBox.classList.add("target-neutral");
    }
}

                <div style="color:#e2e8f0;">Bull Trap detectada → probable giro bajista</div>
                <div style="color:#94a3b8;font-size:12px;">Barrida arriba + magnet DOWN${trap.compression_confirmed ? " + compresión" : ""}. Confianza ${conf}.</div>
            </div>
            `;
        }

        if (trap.type === "bear_trap") {
            html += `
            <div style="margin-bottom:8px;padding:8px 10px;border-radius:10px;background:rgba(59,130,246,0.18);border:1px solid rgba(59,130,246,0.28);">
                <div style="color:#93c5fd;font-weight:bold;">🪤 ${meta.label} · ${meta.role}</div>

            </div>
            `;
        }
    }

    return html;
}

function renderBias(bias) {
    const el = document.getElementById("biasValue");
    if (!el) return;

    if (!bias) {
        el.textContent = "—";
        return;
    }

    let text = "";

    if (bias.bias === "bullish") {
        text = `Bullish ${bias.bullish_pct}%`;
    } else if (bias.bias === "bearish") {
        text = `Bearish ${bias.bearish_pct}%`;
    } else {
        text = "Neutral";
    }

    if (bias.target !== null && bias.target !== undefined) {
        text += ` → ${bias.target}`;
    }

    el.textContent = text;
}

function renderLiquidity(state) {
    const box = document.getElementById("liquidityBox");
    if (!box) return;

    const frames = ["4h", "1h", "5m", "1m"];
    let html = "";

    for (const tf of frames) {
        const liq = state?.[tf]?.liquidity;
        if (!liq) continue;

        html += `
        <div style="margin-bottom:12px; padding-bottom:10px; border-bottom:1px solid rgba(255,255,255,0.08);">
            <div style="font-weight:bold; margin-bottom:6px;">BTC ${tf.toUpperCase()}</div>
            <div>🔼 Liquidez superior: ${liq.range_high_20}</div>
            <div>🔽 Liquidez inferior: ${liq.range_low_20}</div>
            <div>⬆️ Swing high: ${liq.swing_high}</div>
            <div>⬇️ Swing low: ${liq.swing_low}</div>
        </div>
        `;
    }

    box.innerHTML = html || "Sin datos de liquidez.";
}

function renderDebug(data) {

function makeZoneSeries(length, fromValue, toValue) {
    return {
        from: new Array(length).fill(fromValue),
        to: new Array(length).fill(toValue)
    };
}

function drawChart(canvasId, series, liquidity) {
    if (!Array.isArray(series) || series.length === 0) {
        console.warn("Sin datos para", canvasId);
        return;
    }

    if (charts[canvasId]) {
        charts[canvasId].destroy();
    }

    const prices = series.map(x => x.price);
    const ema21 = series.map(x => x.ema21);
    const ema50 = series.map(x => x.ema50);
    const macd = series.map(x => x.macd ?? 0);
    const hist = series.map(x => x.hist ?? 0);

    const labels = prices.map((_, i) => i);

    const hasLiquidity = !!liquidity;
    const highLine = hasLiquidity ? new Array(prices.length).fill(liquidity.range_high_20) : [];
    const lowLine = hasLiquidity ? new Array(prices.length).fill(liquidity.range_low_20) : [];

    const lastPrice = prices[prices.length - 1];
    const zoneSize = lastPrice * 0.002;

    const upperZone = hasLiquidity
        ? makeZoneSeries(prices.length, liquidity.range_high_20 - zoneSize, liquidity.range_high_20 + zoneSize)
        : null;

    const lowerZone = hasLiquidity
        ? makeZoneSeries(prices.length, liquidity.range_low_20 - zoneSize, liquidity.range_low_20 + zoneSize)
        : null;

    const zoneValues = [];
    if (upperZone) zoneValues.push(...upperZone.from, ...upperZone.to);
    if (lowerZone) zoneValues.push(...lowerZone.from, ...lowerZone.to);

    const minPrice = Math.min(
        ...prices,
        ...ema21,
        ...ema50,
        ...(highLine.length ? highLine : []),
        ...(lowLine.length ? lowLine : []),
        ...(zoneValues.length ? zoneValues : [])
    );

    const maxPrice = Math.max(
        ...prices,
        ...ema21,
        ...ema50,
        ...(highLine.length ? highLine : []),
        ...(lowLine.length ? lowLine : []),
        ...(zoneValues.length ? zoneValues : [])
    );

    const pad = (maxPrice - minPrice) * 0.08 || 100;

    const datasets = [];

    if (upperZone) {
        datasets.push(
            {
                type: "line",
                label: "Upper Liquidity Zone Base",
                data: upperZone.from,
                borderWidth: 0,
                pointRadius: 0,
                fill: false,
                yAxisID: "price"
            },
            {
                type: "line",
                label: "Upper Liquidity Zone",
                data: upperZone.to,
                borderWidth: 0,
                pointRadius: 0,
                fill: "-1",
                backgroundColor: "rgba(96, 165, 250, 0.12)",
                yAxisID: "price"
            }
        );
    }

    if (lowerZone) {
        datasets.push(
            {
                type: "line",
                label: "Lower Liquidity Zone Base",
                data: lowerZone.from,
                borderWidth: 0,
                pointRadius: 0,
                fill: false,
                yAxisID: "price"
            },
            {
                type: "line",
                label: "Lower Liquidity Zone",
                data: lowerZone.to,
                borderWidth: 0,
                pointRadius: 0,
                fill: "-1",
                backgroundColor: "rgba(248, 113, 113, 0.12)",
                yAxisID: "price"
            }
        );
    }

    datasets.push(
        {
            type: "line",
            label: "Price",
            data: prices,
            borderColor: "#38bdf8",
            borderWidth: 2,
            pointRadius: 0,
            yAxisID: "price"
        },
        {
            type: "line",
            label: "EMA21",
            data: ema21,
            borderColor: "#22c55e",
            borderWidth: 1.5,
            pointRadius: 0,
            yAxisID: "price"
        },
        {
            type: "line",
            label: "EMA50",
            data: ema50,
            borderColor: "#f97316",
            borderWidth: 1.5,
            pointRadius: 0,
            yAxisID: "price"
        }
    );

    if (hasLiquidity) {
        datasets.push(
            {
                type: "line",
                label: "Liquidity High",
                data: highLine,
                borderColor: "rgba(96, 165, 250, 0.70)",
                borderWidth: 1.5,
                borderDash: [6, 6],
                pointRadius: 0,
                yAxisID: "price"
            },
            {
                type: "line",
                label: "Liquidity Low",
                data: lowLine,
                borderColor: "rgba(248, 113, 113, 0.70)",
                borderWidth: 1.5,
                borderDash: [6, 6],
                pointRadius: 0,
                yAxisID: "price"
            }
        );
    }

    datasets.push(
        {
            type: "bar",
            label: "MACD Hist",
            data: hist,
            backgroundColor: hist.map(v => (v >= 0 ? "#22c55e88" : "#ef444488")),
            yAxisID: "macd"
        },
        {
            type: "line",
            label: "MACD",
            data: macd,
            borderColor: "#eab308",
            borderWidth: 1.5,
            pointRadius: 0,
            yAxisID: "macd"
        },
        {
            type: "line",
            label: "Upper band",
            data: new Array(prices.length).fill(800),
            borderColor: "#334155",
            borderWidth: 1,
            pointRadius: 0,
            yAxisID: "macd"
        },
        {
            type: "line",
            label: "Lower band",
            data: new Array(prices.length).fill(-800),
            borderColor: "#334155",
            borderWidth: 1,
            pointRadius: 0,
            yAxisID: "macd"
        }
    );

    charts[canvasId] = new Chart(document.getElementById(canvasId), {
        data: {
            labels,
            datasets
        },
        options: {
            responsive: true,
            maintainAspectRatio: false,
            plugins: {
                legend: { display: false }
            },
            scales: {
                price: {
                    type: "linear",
                    position: "left",
                    min: minPrice - pad,
                    max: maxPrice + pad,
                    grid: {
                        color: "rgba(255,255,255,0.05)"
                    },
                    ticks: {
                        color: "#9ca3af"
                    }
                },
                macd: {
                    type: "linear",
                    position: "right",
                    grid: {
                        drawOnChartArea: false
                    },
                    ticks: {
                        color: "#9ca3af"
                    }
                }
            }
        }
    });
}
